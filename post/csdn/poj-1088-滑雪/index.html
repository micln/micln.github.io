POJ 1088 滑雪

Mar 23, 2013 1 分钟阅读时间 usc, csdn
POJ Algorithm ACM

此页面通过工具从 csdn 导出，格式可能有问题。

题目

http://poj.org/problem?id=1088

思路

经典的记忆化搜索：搜索+记忆化
搜索：一个一个找，能滑就滑
记忆化：优化重复的路径，记录每个点能滑的最远距离（不论从哪里滑倒这里，从这里往下滑行的最远距离是一定的）

代码 

#include <cstdio>
const int dx[]={0,0,1,-1};
const int dy[]={1,-1,0,0};
int n,m,a[111][111],f[111][111],max;
int so(int x,int y){
	if (f[x][y]) return f[x][y];
	for (int i(0);i<4;i++){
		int xx=x+dx[i];
		int yy=y+dy[i];
		if (a[xx][yy]>=a[x][y]) continue;
		if (so(xx,yy)+1>f[x][y]) f[x][y]=f[xx][yy]+1;
	}
	return f[x][y];
}
int main(){
	scanf("%d%d",&n,&m);
	for (int i(1);i<=n;i++) a[i][0]=a[i][m+1]=11111;
	for (int i(1);i<=m;i++) a[0][i]=a[n+1][i]=11111;
	for (int i(1);i<=n;i++){
		for (int j(1);j<=m;j++){
			scanf("%d",&a[i][j]);
		}
	}
	for (int i(1);i<=n;i++){
		for (int j(1);j<=m;j++){
			f[i][j]=so(i,j);
			if (f[i][j]>max) max=f[i][j];
		}
	}
	printf("%d\n",max+1);
	return 0;
}
